workflow.ts  ·  abbreviated
const step = (cmd) => async (s) => s.session.query(cmd)
 
defineWorkflow({ name: "ship" })
.for("claude")
.run(async (ctx) => {
await ctx.stage({ name: "scan" }, {}, {}, step("/security-review"))
await ctx.stage({ name: "test" }, {}, {}, step("bun test"))
await ctx.stage({ name: "lint" }, {}, {}, step("/lint-fix"))
await ctx.stage({ name: "review" }, {}, {}, step("/pr-summary"))
})
.compile()
